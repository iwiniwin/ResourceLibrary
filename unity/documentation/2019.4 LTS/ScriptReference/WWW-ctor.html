Obsolete public WWW (string url);

参数

url要下载的 URL。必须经过“%”转义处理。

返回

void 一个新的 WWW 对象。下载后，可以从返回的对象中获取结果。

描述

使用给定的 URL 创建一个 WWW 请求。

该函数创建并发送一个 GET 请求。 流将自动开始下载响应。

创建流后，必须等待其完成，然后才能访问下载的数据。 为方便起见，可以 yield 流，从而轻松地指示 Unity 等待下载完成。

注意： URL 必须经过“%”转义处理。

// Get the latest webcam shot from outside "Friday's" in Times Square

using UnityEngine;
using System.Collections;

public class ExampleClass : MonoBehaviour
{
    public string url = "http://images.earthcam.com/ec_metros/ourcams/fridays.jpg";
    IEnumerator Start()
    {
        // Start a download of the given URL
        using (WWW www = new WWW(url))
        {
            // Wait for download to complete
            yield return www;

            // assign texture
            Renderer renderer = GetComponent<Renderer>();
            renderer.material.mainTexture = www.textureNonReadable;
        }
    }
}
